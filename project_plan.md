## 3. Database Schema (Drizzle)

```typescript
// db/schema.ts
import {
  pgTable, pgEnum, serial, text, integer, boolean,
  timestamp, numeric, index,
} from 'drizzle-orm/pg-core'
import { relations } from 'drizzle-orm'

// ── Enums ──────────────────────────────────────────────────────

export const tipoCancha   = pgEnum('tipo_cancha',   ['futbol5', 'futbol7'])
export const rolUsuario   = pgEnum('rol_usuario',   ['admin', 'staff'])
export const estadoReserva = pgEnum('estado_reserva', ['confirmada', 'pendiente_pago', 'cancelada'])
export const metodoPago   = pgEnum('metodo_pago',   ['efectivo', 'transferencia'])

// ── Tables ─────────────────────────────────────────────────────

export const canchas = pgTable('canchas', {
  id:        serial('id').primaryKey(),
  nombre:    text('nombre').notNull(),
  tipo:      tipoCancha('tipo').notNull(),
  capacidad: integer('capacidad').notNull().default(10),
  orden:     integer('orden').notNull().default(0),  // sort_order for CourtGrid
  activa:    boolean('activa').notNull().default(true),
  createdAt: timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
})

export const clientes = pgTable('clientes', {
  id:        serial('id').primaryKey(),
  nombre:    text('nombre').notNull(),
  telefono:  text('telefono'),
  createdAt: timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
})

export const usuarios = pgTable('usuarios', {
  id:           serial('id').primaryKey(),
  email:        text('email').notNull().unique(),
  passwordHash: text('password_hash').notNull(),
  rol:          rolUsuario('rol').notNull().default('staff'),
  nombre:       text('nombre'),
  createdAt:    timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
})

// Recurring series definition — individual booking rows are materialized from this
export const seriesRecurrentes = pgTable('series_recurrentes', {
  id:          serial('id').primaryKey(),
  canchaId:    integer('cancha_id').notNull().references(() => canchas.id),
  clienteId:   integer('cliente_id').notNull().references(() => clientes.id),
  // dia_semana: 0=Sunday … 6=Saturday (matches JS Date.getDay())
  // Argentine UI should label: 1=Lunes … 0=Domingo
  diaSemana:   integer('dia_semana').notNull(),
  horaInicio:  text('hora_inicio').notNull(),  // "HH:mm" e.g. "21:00"
  horaFin:     text('hora_fin').notNull(),      // "HH:mm" e.g. "22:30"
  precio:      numeric('precio', { precision: 10, scale: 2 }).notNull(),  // ARS
  fechaInicio: timestamp('fecha_inicio', { withTimezone: true }).notNull(),
  fechaFin:    timestamp('fecha_fin',    { withTimezone: true }),  // null = indefinite
  activa:      boolean('activa').notNull().default(true),
  createdAt:   timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
})

export const reservas = pgTable('reservas', {
  id:           serial('id').primaryKey(),
  canchaId:     integer('cancha_id').notNull().references(() => canchas.id),
  clienteId:    integer('cliente_id').notNull().references(() => clientes.id),
  inicio:       timestamp('inicio', { withTimezone: true }).notNull(),  // UTC stored
  fin:          timestamp('fin',    { withTimezone: true }).notNull(),
  estado:       estadoReserva('estado').notNull().default('confirmada'),
  precio:       numeric('precio', { precision: 10, scale: 2 }).notNull(),
  notas:        text('notas'),
  serieId:      integer('serie_id').references(() => seriesRecurrentes.id),  // null for ad-hoc
  esRecurrente: boolean('es_recurrente').notNull().default(false),
  creadoPor:    integer('creado_por').references(() => usuarios.id),
  createdAt:    timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
}, (t) => ({
  // Primary query pattern: calendar for a court on a date
  idxCanchaInicio: index('idx_reservas_cancha_inicio').on(t.canchaId, t.inicio),
  // Bulk operations on a series (cancel all future, etc.)
  idxSerie: index('idx_reservas_serie').on(t.serieId),
}))

export const pagos = pgTable('pagos', {
  id:        serial('id').primaryKey(),
  reservaId: integer('reserva_id').notNull().references(() => reservas.id),
  monto:     numeric('monto', { precision: 10, scale: 2 }).notNull(),
  metodo:    metodoPago('metodo').notNull(),
  pagadoEn:  timestamp('pagado_en', { withTimezone: true }).notNull().defaultNow(),
  notas:     text('notas'),
})

// ── Relations (for db.query.* relational API) ──────────────────

export const canchasRelations = relations(canchas, ({ many }) => ({
  reservas: many(reservas),
  series:   many(seriesRecurrentes),
}))

export const reservasRelations = relations(reservas, ({ one, many }) => ({
  cancha:  one(canchas,            { fields: [reservas.canchaId],  references: [canchas.id] }),
  cliente: one(clientes,           { fields: [reservas.clienteId], references: [clientes.id] }),
  serie:   one(seriesRecurrentes,  { fields: [reservas.serieId],   references: [seriesRecurrentes.id] }),
  pagos:   many(pagos),
}))

export const pagosRelations = relations(pagos, ({ one }) => ({
  reserva: one(reservas, { fields: [pagos.reservaId], references: [reservas.id] }),
}))
```

**Recurring bookings: Series + Materialization pattern**

`series_recurrentes` stores the rule only (day of week + time range + price). When a series is created, a Server Action generates and inserts individual `reservas` rows for 52 weeks ahead in a single transaction. Each generated row carries `serie_id` and `es_recurrente = true`. This keeps calendar queries to a single `SELECT` from `reservas` with no special-casing. Individual cancellations update one row's `estado`; bulk future cancellation updates all rows for that `serie_id` where `inicio > NOW()`.

**Timezone handling**: all timestamps stored as UTC (`withTimezone: true`). Display layer uses `Intl.DateTimeFormat` with `timeZone: 'America/Argentina/Buenos_Aires'`. Argentina is permanently UTC-3 (no DST since 2008) so no ambiguity.

**`numeric` caveat**: Drizzle returns `numeric` columns as JavaScript strings. Always parse with `parseFloat()` before arithmetic. Format for display with `formatARS()` utility.

**Conflict detection flow** (inside `createReserva`):

```
db.transaction(async (tx) => {
  const conflicts = await tx
    .select({ id: reservas.id })
    .from(reservas)
    .where(
      and(
        eq(reservas.canchaId, input.canchaId),
        ne(reservas.estado, 'cancelada'),
        lt(reservas.inicio, input.fin),
        gt(reservas.fin, input.inicio),
      )
    )
    .for('update')           // SELECT FOR UPDATE — prevents concurrent inserts

  if (conflicts.length > 0) throw new ConflictError('Horario ya reservado')

  return await tx.insert(reservas).values(input).returning()
})
```

**Turno fijo materialization flow** (inside `createSerie`):

```
1. INSERT series_recurrentes → serieId
2. Compute next 52 occurrences of diaSemana starting from fechaInicio
   using date-fns: eachWeekOfInterval + filter by getDay()
3. For each date: combine with horaInicio/horaFin to get UTC timestamps
4. Batch INSERT reservas (up to 52 rows) with serieId, esRecurrente=true
5. Commit transaction
```

**Layout rules (CSS grid)**

| Court count | Grid columns |
|-------------|-------------|
| 1 | `1fr` (full width) |
| 2 | `1fr 1fr` |
| 3–4 | `1fr 1fr` (wraps to 2 rows) |

**Availability query** — counts non-cancelled bookings per court for the selected date, maps to a color:

```
SELECT cancha_id, COUNT(*) as total_slots,
  COUNT(*) FILTER (WHERE estado != 'cancelada') as ocupados
FROM reservas
WHERE inicio::date = $date
GROUP BY cancha_id
```

- `ocupados = 0` → verde (libre)
- `0 < ocupados < total_possible_slots` → amarillo (parcial)
- `ocupados >= total_possible_slots` → rojo (llena)

Where `total_possible_slots = (24 - 8) hours / slot_duration_hours` for a full day.

**Admin court management** (`/configuracion/canchas`): a list UI (shadcn/ui `Card` + drag-to-reorder via `@dnd-kit/sortable`) that manages only name, type, and order. No coordinates. Each reorder fires `reorderCanchas(ids[])` which updates `orden` fields in a batch.

## 6. Calendar Architecture

**Custom time grid design**

```
         [Cancha 1]  [Cancha 2]  [Cancha 3]    ← CourtHeader row
08:00  │            │            │
08:30  │            │ █ Pérez    │             ← BookingBlock (30-min slot)
09:00  │            │ █ (cont.)  │
09:30  │ █ García   │            │
...
22:00  │ █ Rodríguez│ █ López    │ █ Fijo ★   ← turno fijo badge
22:30  │ (cont.)    │            │ (cont.)
```

**CSS Grid mapping**

```typescript
// Operating range: 08:00–24:00 = 16 hours
// Slot size: 30 min → 32 rows; 60 min → 16 rows; 90 min → ~11 rows
const SLOT_MINUTES = 60
const TOTAL_SLOTS = (24 - 8) * 60 / SLOT_MINUTES  // 16

// Grid: column 1 = time axis, columns 2..N+1 = courts
gridTemplateColumns: `64px repeat(${courts.length}, 1fr)`
gridTemplateRows: `48px repeat(${TOTAL_SLOTS}, 56px)`  // header + slots

// BookingBlock position:
gridColumnStart: courtIndex + 2
gridRowStart: ((startHour - 8) * 60 + startMinute) / SLOT_MINUTES + 2
gridRowEnd: gridRowStart + durationSlots
```

**Conflict prevention**

- **Optimistic UI**: UI prevents selecting an already-occupied time range (cell shows as filled/unclickable).
- **Server validation**: `createReserva` Server Action runs conflict check inside a transaction with `FOR UPDATE` before inserting. If a race condition slips through, the action throws `ConflictError` and the client shows a toast + refreshes the calendar.

**Turno fijo rendering**: `esRecurrente = true` rows render with a distinct background color (e.g., blue tint vs. default green) and a `★` badge. Cancelled individual occurrences render with strikethrough. Visually identical to ad-hoc bookings in shape/position, only color and badge differ.

**Turno fijo materialization strategy: eager, 52 weeks**

Individual booking rows are generated at series creation time. Calendar query remains a simple `SELECT FROM reservas WHERE cancha_id = ? AND inicio::date = ?` with no special-casing. This is valid because:
- 52 rows per series is trivially small
- Conflict detection stays a single query
- Individual cancellation = single `UPDATE`
- Bulk future cancellation = `UPDATE ... WHERE serie_id = ? AND inicio > ?`

## 7. Auth & Role Guards

**`auth.ts` at project root**

```typescript
import NextAuth, { type DefaultSession } from 'next-auth'
import Credentials from 'next-auth/providers/credentials'

// Augment session type to include rol
declare module 'next-auth' {
  interface Session {
    user: DefaultSession['user'] & { rol: 'admin' | 'staff'; id: string }
  }
}

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [Credentials({ /* bcrypt verify against usuarios table */ })],
  callbacks: {
    jwt({ token, user }) {
      if (user) { token.rol = (user as any).rol; token.id = user.id }
      return token
    },
    session({ session, token }) {
      session.user.rol = token.rol as 'admin' | 'staff'
      session.user.id  = token.id as string
      return session
    },
  },
  session: { strategy: 'jwt' },
  pages: { signIn: '/login' },
})
```

**Middleware** — authentication only (not role-based):

```typescript
// middleware.ts — protects everything except /login and Next.js internals
export { auth as middleware } from '@/auth'
export const config = { matcher: ['/((?!api/auth|_next/static|_next/image|login|favicon).*)'] }
```

**Role enforcement in Server Components / Server Actions**

```typescript
// lib/utils.ts
import { auth } from '@/auth'
import { redirect } from 'next/navigation'

export async function requireRole(role: 'admin' | 'staff') {
  const session = await auth()
  if (!session) redirect('/login')
  if (role === 'admin' && session.user.rol !== 'admin') redirect('/')
  return session
}
```

Server Actions call `requireRole('admin')` before any admin-only mutation (court config, revenue reports). Staff-accessible actions call `requireRole('staff')` (which passes for both roles).

## 9. Open Questions & Risks

**Unresolved before building**
- [ ] What is the default slot duration? 60 min or 90 min? Should it be configurable per court or global?
- [ ] Should prices vary by peak hours? If so, how does the booking form determine price — manual entry or auto-calculated from a price table?
- [ ] What is the exact definition of "parcial" for court availability coloring? (e.g., any booking vs. >50% slots taken)
- [ ] Should staff be able to create bookings for past time slots? (useful for walk-ins that weren't pre-booked)
- [ ] How many courts does the venue actually have? (affects performance assumptions but no hard limit is set)

**Hardest technical problems**

| Problem | Risk | Mitigation |
|---------|------|-----------|
| Turno fijo conflict detection | New ad-hoc booking collides with an already-materialized recurring row | Conflict check queries `reservas` table — since recurring rows are materialized, the check is identical to ad-hoc. No special case needed. |
| Timezone display bugs | UTC stored times display wrong in Argentine timezone | Use `Intl.DateTimeFormat` with `timeZone: 'America/Argentina/Buenos_Aires'` consistently. Never use `toLocaleDateString()` without a timezone argument. |
| Concurrent booking race condition | Two staff members book the same slot simultaneously | `SELECT FOR UPDATE` inside Drizzle transaction prevents this. Acceptable for single-venue low-concurrency use. |
| NextAuth v5 beta instability | `next-auth@5.0.0-beta.30` is pre-release | Pin the exact beta version. Do not update during development without testing auth flows. |
| `numeric` ARS parsing | Drizzle returns `numeric` as strings; arithmetic fails silently | Enforce `parseFloat()` in all query result mappers. Add a lint rule or TypeScript branded type `ARS = string` to make it visible. |
| 52-week materialization transaction | If series creation fails mid-way, partial rows may be inserted | Entire operation runs in a single Drizzle transaction — failure rolls back all 52 rows automatically. |

### Phase 8: Performance — Memoization & SSR Optimization

Performance work is scoped to the inflection points identified by static audit. Only apply memoization where there is a measurable or structural reason; do not memoize leaf components or trivial computations.

**Identified inflection points**

| Location | Severity | Issue |
|----------|----------|-------|
| `CourtList.tsx:90-95` | HIGH | `useSensors()` config object recreated on every render, passed to `DndContext` |
| `CourtList.tsx:116` | HIGH | `items.map((c) => c.id)` creates new array on every render for `SortableContext` |
| `CourtList.tsx:86-88` | MEDIUM | `useEffect` syncs `initialCanchas` prop; fires whenever parent passes a new array identity |
| `CourtForm.tsx:66-68` | MEDIUM | `boundAction` closure recreated every render |
| `configuracion/canchas/page.tsx:22-26` | MEDIUM | Inline `<Button>` JSX passed as `trigger` prop — new reference per render |
| `lib/queries/canchas.ts:25` | MEDIUM | `getCanchasConDisponibilidad` calls `getCanchas()` as a sub-call; two separate DB round-trips per page load |
| `lib/actions/canchas.ts:17-18` | LOW | `createCancha` fetches all rows just to compute `MAX(orden)` |

**Rules to apply**
- `useMemo` for derived arrays/objects that are dependencies of context providers or effects (`SortableContext items`, `DndContext sensors`)
- `useMemo` for bound closures that are recreated per render in Client Components (`boundAction`)
- Extract inline JSX props to `const` before JSX return in Server Components — reference is stable within a single render and avoids unnecessary prop churn downstream
- Merge `getCanchas()` sub-call into `getCanchasConDisponibilidad()` as a single SQL query with a LEFT JOIN instead of two round-trips
- Replace `MAX(orden)` scan in `createCancha` with `sql\`SELECT COALESCE(MAX(orden), -1) FROM canchas WHERE activa = true\``
- Use Next.js `cache()` wrapper from `'react'` on `getCanchas()` and `getCanchasConDisponibilidad()` so parallel calls within the same request are deduplicated at the React level (replaces the implicit fetch deduplication that only applies to `fetch()`, not Drizzle)

**Tasks**
- [ ] In `CourtList.tsx`: wrap `useSensors(...)` call result in `useMemo` with empty deps `[]`
- [ ] In `CourtList.tsx`: memoize `items.map((c) => c.id)` as `const sortableIds = useMemo(() => items.map((c) => c.id), [items])`; pass `sortableIds` to `SortableContext`
- [ ] In `CourtForm.tsx`: wrap `boundAction` computation in `useMemo(() => cancha ? editAction.bind(null, cancha.id) : createAction, [cancha])`
- [ ] In `configuracion/canchas/page.tsx`: extract `trigger` to a `const triggerButton = <Button>...</Button>` above the JSX return
- [ ] In `lib/queries/canchas.ts`: refactor `getCanchasConDisponibilidad` to use a single query with LEFT JOIN + GROUP BY instead of calling `getCanchas()` then a second query; eliminates one DB round-trip per home page load
- [ ] In `lib/queries/canchas.ts`: wrap both `getCanchas()` and `getCanchasConDisponibilidad()` with `import { cache } from 'react'` so repeated calls in the same RSC render tree share the result
- [ ] In `lib/actions/canchas.ts` (`createCancha`): replace the `db.select({ orden }).from(canchas)` + JS `Math.max()` with a single `db.select({ max: sql<number>\`COALESCE(MAX(${canchas.orden}), -1)\` }).from(canchas).where(eq(canchas.activa, true))`
- [ ] Verify: after CourtList memoization, dnd-kit drag does not cause SortableContext to re-mount mid-drag
- [ ] Verify: home page shows single DB query for availability in network/server logs (no duplicate `getCanchas` call)
- [ ] Verify: `tsc --noEmit` still passes after all changes
